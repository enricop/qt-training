\section{Composing User Interfaces}

%----------------------------------------------------------------------
\begin{slide}{1696}
\frametitle{Objectives}

\vspace*{1em}
\begin{itemize}
\item Elements are often nested
  \begin{itemize}
  \item one element contains others
  \item manage collections of elements
  \end{itemize}
\item Colors, gradients and images
  \begin{itemize}
  \item create appealing UIs
  \end{itemize}
\item Text
  \begin{itemize}
  \item displaying text
  \item handling text input
  \end{itemize}
\item Anchors and alignment
  \begin{itemize}
  \item allow elements to be placed in an intuitive way
  \item maintain spatial relationships between elements
  \end{itemize}
\end{itemize}

\end{slide}
