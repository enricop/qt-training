\subsection{Common Widgets}

%----------------------------------------------------------------------
\begin{slide}[fragile]{1326}
  \frametitle{Text Widgets}
  \flushedImage{widgets/images/textwidgets}
  \begin{itemize}
  \item \textbf{\iCls{QLabel}}
    \begin{itemize}
    \item[] \begin{cpp}
label = new QLabel("Text", parent);
     \end{cpp}
    \item \texttt{setPixmap( pixmap )} - as content
    \end{itemize}
  \item \textbf{\iCls{QLineEdit}}
    \begin{itemize}
    \item[]
      \begin{cpp}
line = new QLineEdit(parent);        
line->setText("Edit me");
line->setEchoMode(QLineEdit::Password);
connect(line, SIGNAL(textChanged(QString)) ...
connect(line, SIGNAL(editingFinished()) ...
      \end{cpp}
    \item \texttt{setInputMask( mask )} -
      \doc{qlineedit.html\#inputMask-prop}{Input Mask}
    \item \texttt{setValidator( validator )} - \doc{qlineedit.html\#setValidator}{Validator}
   \end{itemize}
  \item \textbf{\iCls{QTextEdit}}
    \begin{itemize}
    \item[]
      \begin{cpp}
edit = new QTextEdit(parent);
edit->setPlainText("Plain Text");        
edit->append("<h1>Html Text</h1>");
connect(edit, SIGNAL(textChanged(QString)) ...
      \end{cpp}
   \end{itemize}
\end{itemize}
\end{slide}

%----------------------------------------------------------------------
\begin{slide}[fragile]{1325}
  \frametitle{Button Widgets}
 \begin{itemize}
  \item \textbf{\iCls{QAbstractButton}}
    \begin{itemize}
    \item Abstract base class of buttons
   \end{itemize}
\flushedImage{widgets/images/buttonwidgets}
  \item \textbf{\iCls{QPushButton}}
    \begin{itemize}
    \item[]
      \begin{cpp}
button = new QPushButton("Push Me", parent);
button->setIcon(QIcon("images/icon.png"));
connect(button, SIGNAL(clicked()) ...
      \end{cpp}
    \item \texttt{setCheckable(bool)} - toggle button
    \end{itemize}
 \item \textbf{\iCls{QRadioButton}}
   \begin{itemize}
   \item[]
     \begin{cpp}
radio = new QRadionButton("Option 1", parent);
     \end{cpp}
   \end{itemize}
  \item \textbf{\iCls{QCheckBox}}
    \begin{itemize}
     \item[] \begin{cpp}
check = new QCheckBox("Choice 1", parent);
     \end{cpp}
  \end{itemize}
  \item \textbf{\iCls{QButtonGroup}} - non-visual button manager
    \begin{itemize}
   \item[]
     \begin{cpp}
group = new QButtonGroup(parent);
group->addButton(button); // add more buttons
group->setExclusive(true);
connect(group, SIGNAL(buttonClicked(QAbstractButton*)) ...
     \end{cpp}
   \end{itemize}
  \end{itemize}
\end{slide}

\begin{slide}[fragile]{1324}
  \frametitle{Value Widgets}
 \begin{itemize}
  \item \textbf{\iCls{QSlider}}
    \begin{itemize}
    \item[]
      \begin{cpp}
slider = new QSlider(Qt::Horizontal, parent);
slider->setRange(0, 99);
slider->setValue(42);
connect(slider, SIGNAL(valueChanged(int)) ...
      \end{cpp}
    \end{itemize}
\flushedImage{widgets/images/valuewidgets}
  \item \textbf{\iCls{QProgressBar}}
    \begin{itemize}
    \item[]
      \begin{cpp}
progress = new QProgressBar(parent);
progress->setRange(0, 99);
progress->setValue(42);
// format: %v for value; %p for percentage
progress->setFormat("%v (%p%)");
    \end{cpp}
   \end{itemize}
 \item \textbf{\iCls{QSpinBox}}
   \begin{itemize}
    \item[]
      \begin{cpp}
spin = new QSpinBox(parent);
spin->setRange(0, 99);
spin->setValue(42);
spin->setSuffix(" USD");
connect(spin, SIGNAL(valueChanged(int)) ...
      \end{cpp}
  \end{itemize}
 \end{itemize}
\end{slide}

\begin{slide}[fragile]{1322}
  \frametitle{Item Widgets}
\flushedImage{widgets/images/itemwidgets}
  \begin{itemize}
  \item \textbf{\iCls{QComboBox}}
    \begin{itemize}
    \item[]
      \begin{cpp}
combo = new QComboBox(parent);        
combo->addItem("Option 1", data);
connect(combo, SIGNAL(activated(int)) ...
QVariant data = combo->itemData(index);
     \end{cpp}
   \item \texttt{setCurrentIndex( index )}
   \end{itemize}

  \item \textbf{\iCls{QListWidget}}
    \begin{itemize}
    \item[]
      \begin{cpp}
list = new QListWidget(parent);
list->addItem("Item 1");
// ownership of items with list
item = new QListWidgetItem("Item 2", list);
item->setCheckState(Qt::Checked);        
connect(list, SIGNAL(itemActivated(QListWidgetItem*)) ...
      \end{cpp}
    \item \texttt{QListWidgetItem::setData(Qt::UserRole, data)}

   \end{itemize}
   \medskip
 \item \textit{Other Item Widgets}: \iCls{QTableWidget}, \iCls{QTreeWidget}
 \end{itemize}

\end{slide}
